    echo "Friday"

#!/bin/bash
 
echo "Enter an year"
 
read year
 
if [[ $((year % 100)) -eq 0 ]]
then
	if [[ $((year % 400)) -eq 0 ]]
	then
		echo "$year is a leap year"
	fi
else
	if [[ $((year % 4)) -eq 0 ]]
	then
		echo "$year is a leap year"
	else
		echo "$year is NOT a leap year"
	fi
fi
